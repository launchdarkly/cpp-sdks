68  template <typename FlagOrSegment>
69  void CalculateChanges(

71  SharedCollection<FlagOrSegment> const& existing_flags_or_segments,
72  Collection<FlagOrSegment> const& new_flags_or_segments,
73  DependencySet& updated_items) {
74  for (auto const& old_flag_or_segment : existing_flags_or_segments) {
75  auto new_flag_or_segment =
76  new_flags_or_segments.find(old_flag_or_segment.first);
77  if (new_flag_or_segment != new_flags_or_segments.end() &&
78  new_flag_or_segment->second.version <=
79  old_flag_or_segment.second->version) {
80  continue;

83  // Deleted.

85  kind, old_flag_or_segment.first, updated_items);
86  }
87 
88  for (auto const& flag_or_segment : new_flags_or_segments) {
89  auto oldItem =
90  existing_flags_or_segments.find(flag_or_segment.first);
91  if (oldItem != existing_flags_or_segments.end() &&
92  flag_or_segment.second.version <= oldItem->second->version) {
93  continue;
94  }
95 
96  // Updated or new.
97  dependency_tracker_.CalculateChanges(kind, flag_or_segment.first,
98  updated_items);
99  }
