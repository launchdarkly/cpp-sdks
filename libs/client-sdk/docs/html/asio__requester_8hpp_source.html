35 static bool NeedsRedirect(HttpResult const& res) {
36  // 300, multiple choices. Not actionable.
37  // 302, found, but not available for unforeseen reasons is not actionable.
38  // 303, attempting to change the method. We only want the original method.
39  // Redirect from a PUT to a GET for instance.
40  // 304, for use with etags, needs to be handled by the caller.
41  // 307, same as 303, but for non-GET operations.
42  return res.Status() == 301 ||
43  res.Status() == 308 && res.Headers().count("location") != 0;
44 }

60 static http::request<http::string_body> MakeBeastRequest(
61  HttpRequest const& request) {
62  http::request<http::string_body> beast_request;
63 
64  beast_request.method(ConvertMethod(request.Method()));
65  auto body = request.Body();
66  if (body) {
67  beast_request.body() = body.value();
68  } else {
69  beast_request.body() = "";
70  }
71  if (request.Path().empty()) {
72  beast_request.target("/");
73  } else {
74  beast_request.target(request.Path());
75  }
76 
77  beast_request.prepare_payload();
78  beast_request.set(http::field::host, request.Host());
79 
80  auto const& properties = request.Properties();
81 
82  for (auto const& pair : request.Properties().BaseHeaders()) {
83  beast_request.set(pair.first, pair.second);
84  }
85 
86  beast_request.set(http::field::user_agent, properties.UserAgent());
87 
88  return beast_request;

91 static std::optional<HttpRequest> MakeRedirectRequest(HttpRequest const& req,
92  HttpResult const& res) {
93  auto location = res.Headers().find("location");
94  // Location should be verified to be present before attempting to
95  // make the redirect request.
96  assert(location != res.Headers().end());
97  // Start the request over with the new URL.
98  if (IsAbsolute(location->second)) {
99  return HttpRequest(location->second, req.Method(), req.Properties(),
100  req.Body());
101  }
102  auto new_url = AppendUrl(req.Url(), location->second);
103  if (new_url) {
104  return HttpRequest(*new_url, req.Method(), req.Properties(),
105  req.Body());
106  }
107 
108  return std::nullopt;
109 }

260  template <typename CompletionToken>
261  auto Request(HttpRequest request, CompletionToken&& token) {
262  // TODO: Clang-tidy wants to pass the request by reference, but I am not
263  // confident that lifetime would make sense.

289  void InnerRequest(boost::asio::any_io_executor exec,
290  std::optional<HttpRequest> request,
291  std::function<void(HttpResult)> callback,
292  unsigned char redirect_count) {
293  if (redirect_count > kRedirectLimit) {
294  boost::asio::post(exec, [callback, request]() mutable {
295  callback(
296  HttpResult("Redirects exceeded 20, cancelling request."));
297  });
298  return;
299  }
300 
301  // The request is invalid and cannot be made, so produce an error
302  // result.
303  if (!request || !request->Valid()) {
304  boost::asio::post(exec, [callback, request]() mutable {
305  callback(HttpResult(
306  "The request was malformed and could not be made."));
307  });
308  return;
309  }
310 
311  boost::asio::post(exec, [exec, callback, request, this,
312  redirect_count]() mutable {
313  auto beast_request = MakeBeastRequest(*request);
314 
315  const auto& properties = request->Properties();
316 
317  std::string service =
318  request->Port().value_or(request->Https() ? "https" : "http");
319 
320  std::shared_ptr<ssl::context> ssl;
321  if (service == "https") {
322  ssl = this->ssl_ctx_;
323  }
324 
325  std::make_shared<FoxyClient>(
326  exec, std::move(ssl), request->Host(), service, beast_request,
327  properties.ConnectTimeout(), properties.ResponseTimeout(),
328  [exec, callback, request, this, redirect_count](auto res) {
329  NeedsRedirect(res)
330  ? InnerRequest(exec, MakeRedirectRequest(*request, res),
331  callback, redirect_count + 1)
332  : callback(res);
333  })
334  ->Run();
335  });
336  }
